        async function fetchFollowerCounts() {

            const platforms = [

                { name: 'Facebook', url: 'https://www.facebook.com/StandWithUs' },

                { name: 'Instagram', url: 'https://www.instagram.com/standwithus/' },

                { name: 'Twitter', url: 'https://twitter.com/StandWithUs' },

                { name: 'Threads', url: 'https://www.threads.net/@standwithus' },

                { name: 'YouTube', url: 'https://www.youtube.com/StandWithUs' },

            ];


            const results = document.getElementById('results');

            results.innerHTML = '<p>Fetching follower counts...</p>';


            // Note: This example assumes a backend API to fetch follower counts exists.

            // Replace the following simulated fetch process with actual API calls.


            try {

                const counts = await Promise.all(platforms.map(async platform => {

                    // Simulate fetching data

                    const data = await response.json();

                    return { platform: platform.name, count: data.followerCount };

                }));


                results.innerHTML = counts.map(c => `<p>${c.platform}: ${c.count} followers</p>`).join('');

                results.innerHTML = '<p>Error fetching follower counts. Please try again later.</p>';
